

### zod库
<font style="color:rgb(52, 73, 94);">Zod 是一个 TypeScript 优先的模式声明和验证库。</font>

```typescript
import { z } from 'zod'

const schema = z.object({
  email: z.string().email({ message: 'Please enter a valid email.' }),
  password: z.string().min(8, { message: 'Be at least 8 characters long' })
})

export async function createAction(previousState: any, formData: FormData) {
  const validatedFields = schema.safeParse({
    email: formData.get('email'),
    password: formData.get('password')
  })
  if (!validatedFields.success) {
    return {
      errors: validatedFields.error.flatten().fieldErrors,
    }
  } 
}
```



### 阻止默认的提交行为
```typescript
const handleSubmit = (e: React.FormEvent) => {
  e.preventDefault()
  startTransition(() => {
    formAction(new FormData(e.currentTarget as HTMLFormElement))
  })
}
```
