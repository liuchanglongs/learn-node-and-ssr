async function createAction(formData: FormData) {
  'use server'
}
<form action={createAction}></form>

### 携带更多参数

方案一：通过`<input type="hidden" name="userId" value={userId} />`进行传递

方案二：通过`const createActionWithId = createAction.bind(null, userId)`进行传递

```js
export default async function Page() {
  const userId = 123;
  const role = "admin";
  // server actions
  async function createAction(
    userId: number,
    role: string,
    formData: FormData,
  ) {
    "use server";
    const username = formData.get("username");
    console.log(username, userId, role);
  }
  // 服务端已有、不想暴露给前端的 ID（userId、内部 role） 通过 bind 绑定
  const createActionWithId = createAction.bind(null, userId, role);
  return (
    <div>
      hello page 06
      <form action={createActionWithId}>
        <input className="border" type="text" name="username" />
        <button type="submit">submit</button>
      </form>
    </div>
  );
}
```

:::info
注意：客户端组件的form表单action属性，可以接收异步函数也可以接收同步函数。
服务端只能是异步
:::

```js
"use client";
import { createAction } from "./actions";
export default function Page() {
  const handleKeyDown = (e: React.KeyboardEvent<HTMLInputElement>) => {
    if (e.key === "Enter") {
      e.preventDefault();
      e.currentTarget.form?.requestSubmit();
    }
  };
  function localAction(formData: FormData) {
    // 浏览器中打印，自己发请求处理
    console.log("client", formData.get("username"));
  }
  return (
    <div>
      hello page 06
      {/* <form action={createAction}> */}
      <form action={localAction}>
        <input
          className="border"
          type="text"
          name="username"
          onKeyDown={handleKeyDown}
        />
        <button type="submit">submit</button>
      </form>
    </div>
  );
}
```
